#include <iostream>
#include <fstream>

using namespace std;

int main()
{   
    fstream MyFile;

    MyFile.open("MyFile.txt", ios::out | ios::app); //append mode (don't delete what was in the file already and create the new text with the old one)

    if(MyFile.is_open())
    {
        MyFile << "Hi, This is the first line" << endl;
        MyFile << "Hi, This is the Second line" << endl;
        MyFile << "Hi, This is the Third line" << endl;

        MyFile.close();
    }

    return 0;
}
