#include <iostream>
#include <string>
#include <fstream>

using namespace std;

void CreatFileContent(string FileName);
void printFileContent(string FileName);

int main()
{   
    CreatFileContent("Test");
    printFileContent("Test");

    return 0;
}

void printFileContent(string FileName)
{
    fstream Myfile;

    Myfile.open(FileName, ios::in);

    if(Myfile.is_open())
    {
        string line;
        while(getline(Myfile, line))
        {
            cout << line << endl;
        }

        Myfile.close();
    }
}

void CreatFileContent(string FileName)
{
    fstream MyFile;

    MyFile.open(FileName, ios::out | ois::app); //append mode (don't delete what was in the file already and create the new text with the old one)

    if(MyFile.is_open())
    {
        MyFile << "My" << endl;
        MyFile << "Name is" << endl;
        MyFile << "Muwafaa" << endl;

        MyFile.close();
    }
}
