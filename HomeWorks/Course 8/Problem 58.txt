#include <iostream>
#include <ctime>
#include <string>

using namespace std;

struct sDate
{
    short year;
    short month;
    short day;
};

struct sPeriod
{
    sDate startDate;
    sDate endDate;
};

enum enCompareDateValue {Before = -1, Equal = 0, After = 1};

short ReadYear(); 
short ReadMonth();
short ReadDay();
sDate readAllDate();

bool IsDateOneBeforeDateTwo(sDate date1, sDate date2);
bool IsDatesEqual(sDate date1, sDate date2);

bool IsOverLapPeriods(sPeriod period1, sPeriod period2);

int main()
{
    sPeriod period1, period2; 

    cout << "\nEnter First Period:\n" << endl;
    cout << "Enter Start Date:" << endl;
    period1.startDate = readAllDate();
    cout << "Enter End Date:" << endl;
    period1.endDate   = readAllDate();

    cout << "\nEnter Seond Period:\n" << endl;
    cout << "Enter Start Date:" << endl;
    period2.startDate = readAllDate();
    cout << "Enter End Date:" << endl;
    period2.endDate   = readAllDate();

    if(IsOverLapPeriods(period1, period2))
        cout << "YES, Periods OverLap" << endl;
    else
        cout << "NO, Periods DOESN'T OverLap" << endl;
    return 0;
}

short ReadYear()
{
    short year = 0;
    cout << "Please Enter a year: ";
    cin >> year;
    return year;
}

short ReadMonth()
{
    short month = 0;
    cout << "Please Enter a month: ";
    cin >> month;
    return month;
}

short ReadDay()
{
    short day = 0;
    cout << "Please enter a Day? "; 
    cin >> day;
    return day;
}

sDate readAllDate()
{
    sDate date;
    date.day   = ReadDay();
    date.month = ReadMonth();
    date.year  = ReadYear();
    cout << "\n";
    return date;
}

bool IsDateOneBeforeDateTwo(sDate date1, sDate date2)
{
    return (date1.year < date2.year) ? true : 
           ((date1.year == date2.year) ? (date1.month < date2.month ? true : 
           (date1.month == date2.month ? date1.day < date2.day : false)) : false);
}

bool IsDatesEqual(sDate date1, sDate date2)
{   
    return (date1.year == date2.year) && 
           (date1.month == date2.month) && 
           (date1.day == date2.day);
}

enCompareDateValue CompareDate(sDate date1, sDate date2)
{
    if(IsDateOneBeforeDateTwo(date1, date2))
        return enCompareDateValue::Before;
    if(IsDatesEqual(date1, date2))
        return enCompareDateValue::Equal;

    return enCompareDateValue::After;
}

bool IsOverLapPeriods(sPeriod period1, sPeriod period2)
{
    if(CompareDate(period2.endDate, period1.startDate) == enCompareDateValue::Before ||
       CompareDate(period2.startDate, period1.endDate) == enCompareDateValue::After)
       return false;
    else
        return true;
}
