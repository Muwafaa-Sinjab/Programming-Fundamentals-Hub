#include <iostream>
#include <string>
using namespace std;

int ReadPositiveNum(string Massage){
    
    int Number = 0;
    do{
        
        cout << Massage;
        cin >> Number;
        
    }while (Number < 0);
    
    return Number;
}

int PrintReversDigits(int Number){
    
    int Reminder = 0;
    int Reverse = 0;
    
    while (Number > 0){
        
        Reminder = Number % 10;
        Number = Number / 10;
        Reverse = Reverse * 10 + Reminder;
    }
    return Reverse;
}

void PrintResult(int Number, int Reverse){
    
    if (Number == Reverse)
        cout << "\nYes, it is a Palindrome number." << endl;
    else
        cout << "\nNo, it is NOT a Palindrome number.\n";
}

int main()
{
    int Number = ReadPositiveNum("Enter Positive Number: ");
    int Reverse = PrintReversDigits(Number);
    PrintResult(Number, Reverse);
    return 0;
}
