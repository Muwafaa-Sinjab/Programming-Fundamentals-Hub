#include <iostream>
#include <string>
using namespace std;

int ReadPositiveNum(string Massage){
    
    int Number = 0;
    do{
        
        cout << Massage;
        cin >> Number;
        
    }while (Number < 0);
    
    return Number;
}

void PrintInvertedPatternNum(int Number){
    
    for (int i = 1; i <= Number; i++) 
    {
        for (int j = 1; j <= i; j++) 
        {
            cout << i;
        }
        
        cout << endl; 
    }
}

int main()
{
    PrintInvertedPatternNum(ReadPositiveNum("Enter Positive Number: "));
    return 0;
}
