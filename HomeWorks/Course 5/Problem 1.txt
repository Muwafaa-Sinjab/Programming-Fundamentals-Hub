#include <iostream>
using namespace std;

void MultiplicationHeader(){
    
    cout << "\n\n\t\t\t Multiplication Table From 1 to 10\n" << endl;
    
    for (int i = 1; i <= 10; i++){
       
       cout << "\t" << i;
    }
    
    cout << "\n-----------------------------------------------------------------------------------\n";
    
}

string ColumSperator(int i){
    
    if (i < 10)
        return "   |";
    else 
        return "  |";
    
}

void MultiplicationTable(){
    
    MultiplicationHeader();
    
    for (int i = 1; i <= 10; i++){
        
        cout << "\n" << i << ColumSperator(i) << "\t";
    
        for (int j = 1; j <= 10; j++){
            
            cout << i * j << "\t";
        }
        
    }
    
}

int main()
{
	MultiplicationTable();
    return 0;
}
