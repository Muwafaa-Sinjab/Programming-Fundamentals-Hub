#include <iostream>
using namespace std;

float ReadNumber()
{
    float Number;
    cout << "Enter a Number: ";
    cin >> Number;
    return Number;
}

float MyABS(int Number)
{
    if (Number > 0)
        return Number;
    else
        return Number * -1;
}

int main()
{
    float Number = ReadNumber();
    cout << "My abs result = " << MyABS(Number) << endl;
    cout << "C++ abs result = " << abs(Number) << endl;

    return 0;
}
