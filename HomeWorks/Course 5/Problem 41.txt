bool IsPalindromeArray(int Arr[100], int ArrLength)
{
    for (int i = 0; i < ArrLength; i++)
    {
        if (Arr[i] != Arr[ArrLength - i  - 1])
        {
            return false;
        }
    }
    return true;
}
