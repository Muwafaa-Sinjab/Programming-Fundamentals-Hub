string readString(string message)
{
    string input;
    cout << message;
    getline(cin,input);
    return input;
}

bool IsVowelCharacter(char ch)
{
    char vowels[] = {'a', 'e', 'i', 'o', 'u'};
    ch = tolower(ch);

    for (char v : vowels)
    {
        if (ch == v)
        {
            return true;
        }
    }
    return false;  
}

void printVowels(string st)
{
    for(int i = 0; i < st.length(); i++)
    {
        if(IsVowelCharacter(st[i]))
            cout << setw(2) << st[i] << "  ";
    }
}
