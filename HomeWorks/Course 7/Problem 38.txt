#include <iostream>
#include <vector>

using namespace std;

string ReadString(string message);
string TrimLeft(string Text);
string TrimRight(string Text);
string Trim(string Text);


int main()
{
    string inputText = ReadString("Please Enter a String: ");

    cout << "String = " << inputText << endl;
    cout << "Trim left = " << TrimLeft(inputText) << endl;
    cout << "Trim right = " << TrimRight(inputText) << endl;
    cout << "Trim = " << Trim(inputText) << endl; 
    return 0;
}

string ReadString(string message)
{
    string inputText;

    cout << message;
    getline(cin, inputText);

    return inputText;
}

string TrimLeft(string Text)
{
    for(int i = 0; i < Text.length(); i++)
    {
        if(Text[i] != ' ')
        {
            Text.erase(0, Text.length() - 1);
            return Text;
        }
    }
    return "";
}
    
string TrimRight(string Text)
{
    for (int i = Text.length() - 1; i >= 0; i--)
    {
        if (Text[i] != ' ')
        {
            Text.erase(0, i -1);
            return Text;
        }
    }
    return "";
}

string Trim(string Text)
{
    Text = TrimLeft(Text);
    Text = TrimRight(Text);

    return Text;
}
