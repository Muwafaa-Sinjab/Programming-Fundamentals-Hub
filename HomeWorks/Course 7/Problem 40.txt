string JoinString(vector<string> inputText, string delimiter)
{
    string text = "";

    for(string& word : inputText)
    {
        text = text + word + delimiter;
    }

    if(!text.empty())
    {
        text = text.substr(0, text.length() - delimiter.length());
    }
    
    return text;
}

string JoinString(string ArrinputText[], short size, string delimiter)
{
    string text = "";

    for(int i = 0; i < size; i++)
    {
        text = text + ArrinputText[i] + delimiter;
    }

    if(!text.empty())
    {
        text = text.substr(0, text.length() - delimiter.length());
    }
    
    return text;
}
