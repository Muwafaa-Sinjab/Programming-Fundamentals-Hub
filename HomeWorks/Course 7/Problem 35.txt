void printString(string st)
{
    string delim = " ";
    short pos = 0;
    string sWord;

    while((pos = st.find(delim)) != std::string::npos)
    {
        sWord = st.substr(0, pos);

        if(sWord != "")
        {
            cout << sWord << endl;
        }
            st.erase(0, pos + delim.length());
    }

    if(st != "")
    {
        cout << st << endl;
    }
}
