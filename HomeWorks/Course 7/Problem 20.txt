#include <iostream>
#include <iomanip>

using namespace std;

int RandomNumber(int From, int To);
void Print2DArray(int Matrix1[3][3], int Rows, int Cols);
bool IsPalindromeMatrix(int Matrix1[3][3], int Rows, int Cols);

int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3] = {{1, 0, 9}, {2, 0, 2}, {3, 4, 3}};
    
    cout << "\nMatrix 1:\n";
    Print2DArray(Matrix1, 3,3);

    if(IsPalindromeMatrix(Matrix1, 3, 3))
        cout << "\nYES: It's Palindrom." << endl;
    else
        cout << "\nNO: It's NOT Palindrom." << endl;
}

void Print2DArray(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

bool IsPalindromeMatrix(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for(int j = 0; j < Cols / 2; j++)
        {
            if(Matrix1[i][j] != Matrix1[i][Cols - 1 - j])
            return false;
        }
        
    }
    return true;
}

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}
