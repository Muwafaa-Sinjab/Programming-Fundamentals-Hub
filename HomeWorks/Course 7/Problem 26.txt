string readString()

    string sentence = "";

    getline(cin, sentence);

string toUpperCase(string sentence)
{   
    for (short i = 0; i < sentence.length(); i++)
    {   
        if(sentence[i] != ' ')
        {
            sentence[i] = toupper(sentence[i]);
        }
    }

    return sentence;
}

string toLowerCase(string sentence)
{   
    for (short i = 0; i < sentence.length(); i++)
    {   
        if(sentence[i] != ' ')
        {
            sentence[i] = tolower(sentence[i]);
        }
    }

    return sentence;
}
