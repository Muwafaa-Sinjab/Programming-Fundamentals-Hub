#include <iostream>
#include <iomanip>
#include <cstdlib>
#include <ctime>

using namespace std;

int RandomNumber(int From, int To);
void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols);
bool AreMatricesEqualBySum(int Matrix1[3][3], int Matrix2[3][3], short Rows, short Cols);
int SumOfMatrix(int Matrix1[3][3], short Rows, short Cols);
void Print2DArray(int Matrix1[3][3], short Rows, short Cols);

int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3];
    int Matrix2[3][3];

    Fill2DArrayWithRandomNumbers(Matrix1, 3, 3);
    cout << "\nMatrix 1:\n";
    Print2DArray(Matrix1, 3, 3);

    Fill2DArrayWithRandomNumbers(Matrix2, 3, 3);
    cout << "\nMatrix 2:\n";
    Print2DArray(Matrix2, 3, 3);

    if (AreMatricesEqualBySum(Matrix1, Matrix2, 3, 3))
        cout << "\nYes, Matrices have the SAME SUM." << endl;
    else
        cout << "\nNO, Matrices are NOT equal." << endl;

    return 0;
}

void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        for (int j = 0; j < Cols; j++)
            Matrix1[i][j] = RandomNumber(1, 10);
}

void Print2DArray(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << setw(2) << setfill('0') << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

bool AreMatricesEqualBySum(int Matrix1[3][3], int Matrix2[3][3], short Rows, short Cols)
{
    int sum1 = SumOfMatrix(Matrix1, Rows, Cols);
    int sum2 = SumOfMatrix(Matrix2, Rows, Cols);

    return sum1 == sum2;
}

int SumOfMatrix(int Matrix1[3][3], short Rows, short Cols)
{
    int Sum = 0;
        for (int i = 0; i < Rows; i++)
            for (int j = 0; j < Cols; j++)
                 Sum += Matrix1[i][j];

        return Sum;
}

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}
