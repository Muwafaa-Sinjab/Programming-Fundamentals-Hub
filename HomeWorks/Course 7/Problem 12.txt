#include <iostream>
#include <iomanip>
#include <cstdlib>
#include <ctime>

using namespace std;

int RandomNumber(int From, int To);
void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols);
bool AreMatricesEqualByTypical(int Matrix1[3][3], int Matrix2[3][3], short Rows, short Cols);
void Print2DArray(int Matrix1[3][3], short Rows, short Cols);

int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3];
    int Matrix2[3][3];

    Fill2DArrayWithRandomNumbers(Matrix1, 3, 3);
    cout << "\nMatrix 1:\n";
    Print2DArray(Matrix1, 3, 3);

    Fill2DArrayWithRandomNumbers(Matrix2, 3, 3);
    cout << "\nMatrix 2:\n";
    Print2DArray(Matrix2, 3, 3);

    if (AreMatricesEqualByTypical(Matrix1, Matrix2, 3, 3))
        cout << "\nYes, Matrices are equal." << endl;
    else
        cout << "\nNO, Matrices are NOT equal." << endl;

    return 0;
}

void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        for (int j = 0; j < Cols; j++)
            Matrix1[i][j] = RandomNumber(1, 10);
}

void Print2DArray(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << setw(2) << setfill('0') << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

bool AreMatricesEqualByTypical(int Matrix1[3][3], int Matrix2[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        for (int j = 0; j < Cols; j++)
            if (Matrix1[i][j] != Matrix2[i][j])
                return false;

    return true;
}

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}
