/* Q: Write a program that includes multiple functions to perform the following tasks: 

1. My_Card_Info:  
   - Prints personal information such as name, age, city, country, monthly salary, yearly salary, gender, and marital status.  

2. SquareOfStars:  
   - Prints a 4x6 square of stars (`*`).  

3. MyfirstProgram:  
   - Prints a motivational message about learning programming.  

4. H_byStars:  
   - Prints the letter **H** using stars (`*`).  

5. In the `main` function:  
   - Call each function in sequence and separate the outputs using a line of asterisks (`***********************`).  

    Notes:  
    - The program is structured using functions for better readability and reusability.  
    - Each function is responsible for a specific task.

*/

#include <iostream>
#include <string>
using namespace std;

void H_byStars() {
    cout << "*\t*\n*\t*\n*********\n*\t*\n*\t*" << endl;
}

void MyfirstProgram() {
    cout << "I love programming!\n" << endl;
    cout << "I promise to be the best developer ever!\n" << endl;
    cout << "I know it will take some time to practice, but I will achieve my goal.\n" << endl;
    cout << "Best Regards;\nMuwafaa Sinjab." << endl;
}

void SquareOfStars() {
    cout << "******\n******\n******\n******" << endl;
}

void My_Card_Info() {
    string MyName = "Muwafaa Sinjab";
    int MyAge = 21;
    string MyCity = "Ankara";
    string MyCountry = "Turkey";
    int MonthSalary = 1000;
    int YearSalary = MonthSalary * 12;
    char Gender = 'M';
    int WifesNumber = 0;

    cout << "Name: " << MyName << endl;
    cout << "Age: " << MyAge << endl;
    cout << "City: " << MyCity << endl;
    cout << "Country: " << MyCountry << endl;
    cout << "Monthly Salary: " << MonthSalary << endl;
    cout << "Yearly Salary: " << YearSalary << endl;
    cout << "Gender: " << Gender << endl;
    cout << "Married: " << WifesNumber << endl;
}

int main() {

    cout << "Task 1\n" << endl;
    My_Card_Info();
    cout << "\n***********************\n\n";

    cout << "Task 2\n" << endl;
    SquareOfStars();
    cout << "\n***********************\n\n";

    cout << "Task 3\n" << endl;
    MyfirstProgram();
    cout << "\n***********************\n\n";

    cout << "Task 4\n" << endl;
    H_byStars();

}
