#include <iostream>
#include <ctime>
#include <string>

using namespace std;

struct sDate
{
    short year;
    short month;
    short day;
};

enum enCompareDateValue {Before = -1, Equal = 0, After = 1};

short ReadYear(); 
short ReadMonth();
short ReadDay();
sDate readAllDate();

bool IsDateOneBeforeDateTwo(sDate date1, sDate date2);
bool IsDatesEqual(sDate date1, sDate date2);
bool IsDateOneAfterDateTwo(sDate date1, sDate date2);

enCompareDateValue CompareDate(sDate date1, sDate date2);

int main()
{
    sDate date1, date2;

    cout << "\nEnter Date1:" << endl;
    date1 = readAllDate();
    cout << "\nEnter Date2:" << endl;
    date2 = readAllDate();

    cout << "Compare Result = " << CompareDate(date1, date2);
    return 0;
}

short ReadYear()
{
    short year = 0;
    cout << "Please Enter a year: ";
    cin >> year;
    return year;
}

short ReadMonth()
{
    short month = 0;
    cout << "Please Enter a month: ";
    cin >> month;
    return month;
}

short ReadDay()
{
    short day = 0;
    cout << "Please enter a Day? "; 
    cin >> day;
    return day;
}

sDate readAllDate()
{
    sDate date;
    date.day   = ReadDay();
    date.month = ReadMonth();
    date.year  = ReadYear();
    cout << "\n";
    return date;
}

bool IsDateOneBeforeDateTwo(sDate date1, sDate date2)
{
    return (date1.year < date2.year) ? true : 
           ((date1.year == date2.year) ? (date1.month < date2.month ? true : 
           (date1.month == date2.month ? date1.day < date2.day : false)) : false);
}

bool IsDatesEqual(sDate date1, sDate date2)
{   
    return (date1.year == date2.year) && 
           (date1.month == date2.month) && 
           (date1.day == date2.day);
}

bool IsDateOneAfterDateTwo(sDate date1, sDate date2)
{
    return (!IsDateOneBeforeDateTwo(date1, date2) && !IsDatesEqual(date1, date2));
}

enCompareDateValue CompareDate(sDate date1, sDate date2)
{
    if(IsDateOneBeforeDateTwo(date1, date2))
        return enCompareDateValue::Before;
    if(IsDatesEqual(date1, date2))
        return enCompareDateValue::Equal;

    return enCompareDateValue::After;
}
