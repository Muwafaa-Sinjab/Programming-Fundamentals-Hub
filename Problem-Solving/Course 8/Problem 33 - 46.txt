sDate readAllDate()
{
    sDate date;

    date.day   = ReadDay();
    date.month = ReadMonth();
    date.year  = ReadYear();

    return date;
}

bool IsleapYear(short year)
{
    return (year % 4 == 0 && year % 100 != 0 ) || (year % 400 == 0);
}

short NumberOfDaysInMonth(short year, short month)
{
    if (month < 1 || month > 12) return 0;

    short arrNumberOfDays[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    return (month == 2) ? (IsleapYear(year) ? 29 : 28) : arrNumberOfDays[month - 1];
}

sDate DecreaseDateByOneDay(sDate date)
{
    if(date.month == 1 && date.day == 1)
    {
        date.year--;
        date.month = 12;
        date.day = NumberOfDaysInMonth(date.year, date.month);
    }
    else if(date.day == 1)
    {
        date.month--;
        date.day = NumberOfDaysInMonth(date.year, date.month);
    }else
    {
        date.day--;
    }
    return date;
}

sDate DecreaseDateByXDays(short xDays, sDate date)
{
    for(int i = 0; i < xDays; i++)
    {
        date = DecreaseDateByOneDay(date);
    }
    return date;
}

sDate DecreaseDateByOneWeek(sDate date)
{
    for(int i = 0; i < 7; i++)
    {
        date = DecreaseDateByOneDay(date);
    }
    return date;
}

sDate DecreaseDateByXWeeks( short xWeeks, sDate date)
{
    for(int i = 0; i < xWeeks; i++)
    {
        date = DecreaseDateByOneWeek(date);
    }
    return date;
}

sDate DecreaseDateByOneMonth(sDate date)
{
    if(date.month == 1)
    {
        date.month = 12;
        date.year--;
    }
    else
    {
        date.month--;
    }

    short numberOfDaysInCurrentMonth = NumberOfDaysInMonth(date.year, date.month);

    if(date.day > numberOfDaysInCurrentMonth)
    {
        date.day = numberOfDaysInCurrentMonth;
    }
    return date;
}

sDate DecreaseDateByXMonths(short xMonths, sDate date)
{
    for(int i = 0; i < xMonths; i++)
    {
        date = DecreaseDateByOneMonth(date);
    }
    return date;
}

sDate DecreaseDateByOneYear(sDate date)
{
    date.year--;
    if(date.month == 2 && date.day == 29 && !IsleapYear(date.year))
    {
        date.day = 28;
    }
    return date;
}

sDate DecreaseDateByXYears(short xYears, sDate date)
{
    for(int i = 0; i < xYears; i++)
    {
        date = DecreaseDateByOneYear(date);
    }
    return date;
}

sDate DecreaseDateByXYearsFaster(short xYearsFaster, sDate date)
{
    date.year -= xYearsFaster;
    if(date.month == 2 && date.day == 29 && !IsleapYear(date.year))
    {
        date.day = 28;
    }
    return date;
}

sDate DecreaseDateByOneDecade(sDate date)
{
    date = DecreaseDateByXYearsFaster(10, date);
    return date;
}

sDate DecreaseDateByXDecade(short xDecade, sDate date)
{
    for(int i = 0; i < xDecade; i++)
    {
        date = DecreaseDateByOneDecade(date);
    }
    return date;
}

sDate DecreaseateByXDecadeFaster(short xDecadeFaster, sDate date)
{
    date.year -= (xDecadeFaster * 10);
    if(date.month == 2 && date.day == 29 && !IsleapYear(date.year))
    {
        date.day = 28;
    }
    return date;
}

sDate DecreaseDateByOneCentury(sDate date)
{
    for(int i = 0; i < 10; i++)
    {
        date = DecreaseDateByOneDecade(date);
    }
    return date;
}

sDate DecreaseDateByOneMillennium(sDate date)
{
    for(int i = 0; i < 10; i++)
    {
        date = DecreaseDateByOneCentury(date);
    }
    return date;
}
