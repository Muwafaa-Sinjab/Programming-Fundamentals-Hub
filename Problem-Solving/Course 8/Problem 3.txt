#include <iostream>

using namespace std;

short ReadYear(); 
bool IsleapYear(short year);

int main()
{
    short year = ReadYear();

    if(IsleapYear(year))
        cout << year << " is a leap year." << endl;
    else
        cout << year << " is not a leap year." << endl;

    return 0;
}

short ReadYear()
{
    short year = 0;
    cout << "Enter the year: ";
    cin >> year;
    return year;
}

bool IsleapYear(short year)
{
    return (year % 4 == 0 && year % 100 != 0 ) || (year % 400 == 0)
}
