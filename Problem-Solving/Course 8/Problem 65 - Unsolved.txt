#include <iostream>
#include <string>
#include <vector>

using namespace std;

struct sDate
{
    short year;
    short month;
    short day;
};

string ReadString(string message);
vector<string> SplitString(string text, string delimiter);
sDate StringToDate(string dateString);
string FormatDate(sDate date, string format);

int main()
{
    string dateString = ReadString("Please Enter Date dd/mm/yyyy? ");
    sDate date = StringToDate(dateString);

    cout << "\n" << FormatDate(date, "dd/mm/yyyy") << endl;
    cout << FormatDate(date, "yyyy/dd/mm") << endl;
    cout << FormatDate(date, "mm/dd/yyyy") << endl;
    cout << FormatDate(date, "mm-dd-yyyy") << endl;
    cout << FormatDate(date, "dd-mm-yyyy") << endl;
    cout << FormatDate(date, "Day:dd, Month:mm, Year:yyyy") << endl;

    return 0;
}

string ReadString(string message)
{
    string input;
    cout << message;
    getline(cin, input);
    return input;
}

vector<string> SplitString(string text, string delimiter)
{
    short delimiterPosition = 0;
    string currentWord;
    vector<string> words;

    while ((delimiterPosition = text.find(delimiter)) != string::npos)
    {
        currentWord = text.substr(0, delimiterPosition);
        if (currentWord != "")
        {
            words.push_back(currentWord);
        }
        text.erase(0, delimiterPosition + delimiter.length());
    }

    if (text != "")
    {
        words.push_back(text);
    }

    return words;
}

sDate StringToDate(string dateString)
{
    sDate date;
    vector<string> dateData = SplitString(dateString, "/");

    date.day   = stoi(dateData[0]);
    date.month = stoi(dateData[1]);
    date.year  = stoi(dateData[2]);

    return date;
}

string FormatDate(sDate date, string format)
{
    string formattedDateString = "";
    size_t pos;
    
    while ((pos = format.find("dd")) != string::npos) 
        format.replace(pos, 2, to_string(date.day));
        
    while ((pos = format.find("mm")) != string::npos) 
        format.replace(pos, 2, to_string(date.month));
        
    while ((pos = format.find("yyyy")) != string::npos) 
        format.replace(pos, 4, to_string(date.year));

    formattedDateString = format;
    return formattedDateString;
}
