#include <iostream>

using namespace std;

struct sDate
{
    short year;
    short month;
    short day;
};

short ReadYear(); 
short ReadMonth();
short ReadDay();

sDate readAllDate();
bool IsEqualDate(sDate date1, sDate date2);

int main()
{
    sDate date1 = readAllDate();
    cout << "\n";
    sDate date2 = readAllDate();

    if(IsEqualDate(date1, date2))
        cout << "YES, Date1 is Equale to Date2." << endl;
    else
        cout << "NO, Date1 is NOT Equale to Date2." << endl;
    return 0;
}

short ReadYear()
{
    short year = 0;
    cout << "Please Enter to check year: ";
    cin >> year;
    return year;
}

short ReadMonth()
{
    short month = 0;
    cout << "Please Enter to check month: ";
    cin >> month;
    return month;
}

short ReadDay()
{
    short day = 0;
    cout << "Please Enter to check Day: ";
    cin >> day;
    return day;
}

sDate readAllDate()
{
    sDate date;

    date.day   = ReadDay();
    date.month = ReadMonth();
    date.year  = ReadYear();

    return date;
}

bool IsEqualDate(sDate date1, sDate date2)
{   
    return (date1.year == date2.year) && 
           (date1.month == date2.month) && 
           (date1.day == date2.day);
}
