bool IsleapYear(short year)
{
    return (year % 4 == 0 && year % 100 != 0 ) || (year % 400 == 0);
}

short NumberOfDaysInMonth(short year, short month)
{
    if (month < 1 || month > 12) return 0;

    short arrNumberOfDays[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    return (month == 2) ? (IsleapYear(year) ? 29 : 28) : arrNumberOfDays[month - 1];
}

bool IsLastDayInMonth(sDate date)
{
    return (date.day == NumberOfDaysInMonth(date.year, date.month));
}

bool IsDateOneBeforeDateTwo(sDate date1, sDate date2)
{
    return (date1.year < date2.year) ? true : 
           ((date1.year == date2.year) ? (date1.month < date2.month ? true : 
           (date1.month == date2.month ? date1.day < date2.day : false)) : false);
}

bool IsDatesEqual(sDate date1, sDate date2)
{   
    return (date1.year == date2.year) && 
           (date1.month == date2.month) && 
           (date1.day == date2.day);
}

enCompareDateValue CompareDate(sDate date1, sDate date2)
{
    if(IsDateOneBeforeDateTwo(date1, date2))
        return enCompareDateValue::Before;
    if(IsDatesEqual(date1, date2))
        return enCompareDateValue::Equal;

    return enCompareDateValue::After;
}

bool IsOverLapPeriods(sPeriod period1, sPeriod period2)
{
    if(CompareDate(period2.endDate, period1.startDate) == enCompareDateValue::Before ||
       CompareDate(period2.startDate, period1.endDate) == enCompareDateValue::After)
       return false;
    else
        return true;
}

bool IsLastMonthInYear(short month)
{
    return (month == 12);
}

sDate IncreaseDateByOneDay(sDate date)
{
    if(IsLastDayInMonth(date))
    {
        if(IsLastMonthInYear(date.month))
        {
            date.month = 1;
            date.day = 1;
            date.year++;
        }
        else
        {
            date.day = 1;
            date.month++;
        }
    }
    else
    {
        date.day++;
    }
    return date;
}

short GetDifferencesInDays(sDate date1, sDate date2, bool IncludeEndDay)
{
    int Days = 0;
    while (IsDateOneBeforeDateTwo(date1, date2))
    {   
        Days++;
        date1 = IncreaseDateByOneDay(date1);
    }
    return IncludeEndDay ? ++Days : Days;
}

int GetOverlapDaysCount(sPeriod period1, sPeriod period2, bool IncludeEndDay)
{
    if (!IsOverLapPeriods(period1, period2))
        return 0;

    sDate overLapStart, overLapEnd;
    if (IsDateOneBeforeDateTwo(period1.startDate, period2.startDate))
        overLapStart = period2.startDate;
    else
        overLapStart = period1.startDate;

    if (IsDateOneBeforeDateTwo(period1.endDate, period2.endDate))
        overLapEnd = period1.endDate;
    else
        overLapEnd = period2.endDate;
    return (GetDifferencesInDays(overLapStart, overLapEnd, IncludeEndDay));
}
