#include <iostream>

using namespace std;

struct sDate
{
    short year;
    short month;
    short day;
};

short ReadYear(); 
short ReadMonth();
short ReadDay();

sDate readAllDate();
bool IsleapYear(short year);
short NumberOfDaysInMonth(short year, short month);

bool IsLastDayInMonth(sDate date);
bool IsLastMonthInYear(sDate date);

int main()
{
    sDate date = readAllDate();

    if(IsLastDayInMonth(date))
    {
        cout << "\nYES, Day is last Day in month" << endl;
    }
    else
    {
        cout << "\nNO, Day is NOT last Day in month" << endl;
    }

    if(IsLastMonthInYear(date.month))
    {
        cout << "\nYES, Month is last month in Year" << endl;
    }
    else
    {
        cout << "\nNO, Month is NOT last month in Year" << endl;
    }
    
    return 0;
}

short ReadYear()
{
    short year = 0;
    cout << "Please Enter to check year: ";
    cin >> year;
    return year;
}

short ReadMonth()
{
    short month = 0;
    cout << "Please Enter to check month: ";
    cin >> month;
    return month;
}

short ReadDay()
{
    short day = 0;
    cout << "Please Enter to check Day: ";
    cin >> day;
    return day;
}

sDate readAllDate()
{
    sDate date;

    date.day   = ReadDay();
    date.month = ReadMonth();
    date.year  = ReadYear();

    return date;
}

bool IsleapYear(short year)
{
    return (year % 4 == 0 && year % 100 != 0 ) || (year % 400 == 0);
}

short NumberOfDaysInMonth(short year, short month)
{
    if (month < 1 || month > 12) return 0;

    short arrNumberOfDays[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    return (month == 2) ? (IsleapYear(year) ? 29 : 28) : arrNumberOfDays[month - 1];
}

bool IsLastDayInMonth(sDate date)
{   
    return (date.day == NumberOfDaysInMonth(date.year, date.month));
}

bool IsLastMonthInYear(sDate date)
{
    return (date.month == 12);
}
