string LowerFirstLetterOfEachWord(string sentence)
{   
    bool IsFirstLetter = true;

    for (short i = 0; i < sentence.length(); i++)
    {   
        if(sentence[i] != ' ' && IsFirstLetter)
        {
            sentence[i] = tolower(sentence[i]);
        }
        
        IsFirstLetter = (sentence[i] == ' ' ? true : false);
    }

    return sentence;
}
