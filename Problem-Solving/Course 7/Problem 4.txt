#include <iostream>
#include <iomanip>
#include <cstdlib>
#include <ctime>

using namespace std;

int RandomNumber(int From, int To)
{
    int randNum = rand() % (To - From + 1) + From;
    return randNum;
}

void Fill2DArrayWithRandomNumbers(int Arr[3][3], int Rows, int Colm)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Colm; j++)
        {
            Arr[i][j] = RandomNumber(1, 100);
        }
    }
}

void Print2DArray(int Arr[3][3], int Rows, int Colm)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Colm; j++)
        {
            cout << setw(3) << Arr[i][j] << " ";
        }
        cout << endl;
    }
}

int SumOfCols(int Arr[3][3], int ColmNumber, int Rows)
{
    int Sum = 0;

    for (int j = 0; j < Rows; j++)
    {
        Sum += Arr[j][ColmNumber];
    }

    return Sum;
}

void PrintSumOfCols(int Arr[3][3], int Rows, int Colm)
{
    cout << "\nThe following are the sum of each column in matrix:" << endl;

    for (int i = 0; i < Colm; i++)
    {
        cout << "Column " << i + 1 << " sum = "
             << SumOfCols(Arr, i, Rows) << endl;
    }
}

int main()
{
    srand((unsigned)time(NULL));

    int Arr[3][3];

    Fill2DArrayWithRandomNumbers(Arr, 3, 3);

    cout << "The following is a 3x3 Matrix:" << endl;
    Print2DArray(Arr, 3, 3);

    PrintSumOfCols(Arr, 3, 3);

    return 0;
}
