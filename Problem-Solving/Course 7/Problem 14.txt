#include <iostream>

using namespace std;

void Print2DArray(int Matrix1[3][3], int Rows, int Cols);
bool IsScalarMatrix(int Matrix1[3][3], int Rows, int Cols);


int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3] = 
    {
        {2, 0 ,0},
        {0, 2 ,0},
        {0, 0 ,2}
    };
    

    cout << "\nMatrix:\n";
    Print2DArray(Matrix1, 3,3);

    if (IsScalarMatrix(Matrix1, 3, 3))
        cout << "\nYes, the matrix is scalar" << endl;
    else
        cout << "\nNo, the matrix is NOT scalar" << endl;

    return 0;
}

void Print2DArray(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

bool IsScalarMatrix(int Matrix1[3][3], int Rows, int Cols)
{
    if (Rows != Cols)
        return false;

    int diagonalValue = Matrix1[0][0];

    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {   
            if (i == j && Matrix1[i][j] != diagonalValue) return false;
            if (i != j && Matrix1[i][j] != 0) return false;
        }
    }
    return true;
}
