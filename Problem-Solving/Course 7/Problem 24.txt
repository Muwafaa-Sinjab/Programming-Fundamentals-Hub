string CapitalizeFirstLetterOfEachWord(string sentence)
{
    bool isFirstLetter = true;

    for (size_t i = 0; i < sentence.length(); i++)
    {
        if (sentence[i] != ' ' && isFirstLetter)
        {
            sentence[i] = static_cast<char>(
                toupper(static_cast<unsigned char>(sentence[i]))
            );
            isFirstLetter = false;
        }
        else if (sentence[i] == ' ')
        {
            isFirstLetter = true;
        }
    }

    return sentence;
}
