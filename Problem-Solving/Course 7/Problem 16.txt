#include <iostream>
#include <iomanip>

using namespace std;

void Print2DArray(int Matrix1[3][3], int Rows, int Cols);
int CheckCountNumber(int Matrix1[3][3], int Rows, int Cols, int NumberToCheck);
bool IsSparseMatrix(int Matrix1[3][3], int Rows, int Cols);


int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3] = 
    {
        {2, 0 ,0},
        {0, 2 ,0},
        {0, 0 ,2}
    };
    

    cout << "\nMatrix:\n";
    Print2DArray(Matrix1, 3,3);

    if (IsSparseMatrix(Matrix1, 3, 3))
        cout << "YES: It's sparse Matrix" << endl;
    else
        cout << "NO: It's not sparse Matrix" << endl;
    return 0;
}

void Print2DArray(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

int CheckCountNumber(int Matrix1[3][3], int Rows, int Cols, int NumberToCheck)
{
    int Counter = 0;
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {   
            if (Matrix1[i][j] == NumberToCheck)
                Counter++;
        }
    }
    return Counter;
}

bool IsSparseMatrix(int Matrix1[3][3], int Rows, int Cols)
{
    int MatrixSize = (Rows * Cols + 1) / 2;

    return (CheckCountNumber(Matrix1, Rows, Cols, 0) >= MatrixSize);
}
