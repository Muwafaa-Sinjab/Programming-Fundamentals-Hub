#include <iostream>
#include <cctype>

using namespace std;

string readString(string message);
char flipCase(char c);
string flipStringCase(string st);

int main()
{
    string st = readString("Please Enter a String: ");

    st = flipStringCase(st);
    
    cout << "\nString after inverting case:\n" << st << endl;

    return 0;
}

string readString(string message)
{
    string input;
    cout << message << endl;;
    getline(cin, input);

    return input;
}

char flipCase(char c) 
{
    return isupper(c) ? tolower(c) : toupper(c);
}

string flipStringCase(string st) 
{
    for(int i = 0; i < st.length(); i++)
    {
        if(st[i] != ' ')
        {
            st[i] = flipCase(st[i]);
        }
    }
    return st;
}
