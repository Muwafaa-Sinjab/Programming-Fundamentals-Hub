// ===== Function Prototypes =====
int RandomNumber(int From, int To);
void Fill2DArrayWithRandomNumbers(int Arr[3][3], short Rows, short Colm);
void Multiply2DMatrix(int Arr[3][3], int Arr2[3][3], int ArrResults[3][3], short Rows, short Colm);
void Print2DArray(int Arr[3][3], short Rows, short Colm);

// ===== main =====
int main()
{
    srand((unsigned)time(NULL));

    int Arr[3][3], Arr2[3][3], ArrResults[3][3];

    Fill2DArrayWithRandomNumbers(Arr, 3, 3);
    Fill2DArrayWithRandomNumbers(Arr2, 3, 3);

    cout << "\nMatrix 1:" << endl;
    Print2DArray(Arr, 3, 3);

    cout << "\nMatrix 2:" << endl;
    Print2DArray(Arr2, 3, 3);

    Multiply2DMatrix(Arr, Arr2, ArrResults, 3, 3);

    cout << "\nResults:" << endl;
    Print2DArray(ArrResults, 3, 3);

// ===== High / Mid Level Functions =====
void Fill2DArrayWithRandomNumbers(int Arr[3][3], short Rows, short Colm)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Colm; j++)
        {
            Arr[i][j] = RandomNumber(1, 10);
        }
    }
}

void Multiply2DMatrix(int Arr[3][3], int Arr2[3][3], int ArrResults[3][3], short Rows, short Colm)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Colm; j++)
        {
            ArrResults[i][j] = Arr[i][j] * Arr2[i][j];
        }
    }
}

void Print2DArray(int Arr[3][3], short Rows, short Colm)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Colm; j++)
        {
            cout << setw(2) << setfill('0') << Arr[i][j] << "  ";
        }
        cout << endl;
    }
}

// ===== Low Level Function =====
int RandomNumber(int From, int To)
{
    int randNum = rand() % (To - From + 1) + From;
    return randNum;
}
