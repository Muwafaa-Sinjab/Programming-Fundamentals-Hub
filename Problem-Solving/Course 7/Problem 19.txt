#include <iostream>
#include <iomanip>

using namespace std;

int RandomNumber(int From, int To);
void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols);
void Print2DArray(int Matrix1[3][3], int Rows, int Cols);
int MinNumberInMatrix(int Matrix1[3][3], int Rows, int Cols);
int MaxNumberInMatrix(int Matrix1[3][3], int Rows, int Cols);

int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3];
    
    Fill2DArrayWithRandomNumbers(Matrix1, 3, 3);
    cout << "\nMatrix 1:\n";
    Print2DArray(Matrix1, 3,3);

    cout << "\nMinimum Number of matrix is: "
    << MinNumberInMatrix(Matrix1, 3, 3) << endl;

    cout << "\nMaximum Number of matrix is: "
    << MaxNumberInMatrix(Matrix1, 3, 3) << endl;
}

void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        for (int j = 0; j < Cols; j++)
            Matrix1[i][j] = RandomNumber(1, 10);
}

void Print2DArray(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

int MinNumberInMatrix(int Matrix1[3][3], int Rows, int Cols)
{   
    int Minmum = Matrix1[0][0];

    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (Matrix1[i][j] < Minmum)
                Minmum = Matrix1[i][j];
        }
    }
    return Minmum;
}

int MaxNumberInMatrix(int Matrix1[3][3], int Rows, int Cols)
{
    int Max = Matrix1[0][0];

    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (Matrix1[i][j] > Max)
                Max = Matrix1[i][j];
        }
    }
    return Max;
}

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}
