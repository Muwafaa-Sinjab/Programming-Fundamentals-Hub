#include <iostream>
#include <vector>
#include <string>

using namespace std;

string JoinString(vector<string> inputText, string delimiter);

int main()
{
    vector<string> inputText = {"Ali", "Ahmed", "Muwafaa", "Taj"}; 
    
    cout << "string after join = " << JoinString(inputText, "!!!!");

    return 0;
}

string JoinString(vector<string> inputText, string delimiter)
{
    string text = "";

    for(string& word : inputText)
    {
        text = text + word + delimiter;
    }

    if(!text.empty())
    {
        text = text.substr(0, text.length() - delimiter.length());
    }
    
    return text;
}
