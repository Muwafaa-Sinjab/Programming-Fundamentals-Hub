#include <iostream>
#include <iomanip>

using namespace std;

int RandomNumber(int From, int To);
void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols);
void Print2DArray(int Matrix1[3][3], int Rows, int Cols);
bool IsNumberInMatrix(int Matrix1[3][3], int Rows, int Cols, int NumberToCheck);
void IntersectedNumbers(int Matrix1[3][3], int Matrix2[3][3], int Rows, int Cols);

int main()
{
    srand((unsigned)time(NULL));

    int Matrix1[3][3];
    int Matrix2[3][3];
    
    Fill2DArrayWithRandomNumbers(Matrix1, 3, 3);
    cout << "\nMatrix 1:\n";
    Print2DArray(Matrix1, 3,3);

    Fill2DArrayWithRandomNumbers(Matrix2, 3, 3);
    cout << "\nMatrix 2:\n";
    Print2DArray(Matrix2, 3,3);

    cout << "The intersectd Numbers of these 2 matrices are: ";
    IntersectedNumbers(Matrix1, Matrix2, 3, 3);
    return 0;
}

void Fill2DArrayWithRandomNumbers(int Matrix1[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        for (int j = 0; j < Cols; j++)
            Matrix1[i][j] = RandomNumber(1, 10);
}

void Print2DArray(int Matrix1[3][3], int Rows, int Cols)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
            cout << Matrix1[i][j] << "  ";
        cout << endl;
    }
}

bool IsNumberInMatrix(int Matrix1[3][3], int Rows, int Cols, int NumberToCheck)
{
    for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            if (NumberToCheck == Matrix1[i][j])
                return true;
        }
    }
    return false;
}

void IntersectedNumbers(int Matrix1[3][3], int Matrix2[3][3], int Rows, int Cols)
{   
    int Number;

     for (int i = 0; i < Rows; i++)
    {
        for (int j = 0; j < Cols; j++)
        {
            Number = Matrix1[i][j];
            if (IsNumberInMatrix(Matrix2, Rows, Cols, Number))
            {
                cout << setw(2) << Number << "  ";
            }
        }
    }
}

int RandomNumber(int From, int To)
{
    return rand() % (To - From + 1) + From;
}
