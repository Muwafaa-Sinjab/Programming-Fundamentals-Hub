{
    char input;
    cout << message;
    cin >> input;
    return input;
}

int coutSpecificLetterForBothCase(string st, char ch, bool MatchCase)
{
    int count = 0;

    for (char c : st)
    { 
        if (MatchCase) {
            if (c == ch) count++;
        } else {
            if (toupper(c) == toupper(ch)) count++;
        }
    }
    return count;
}

char flipCase(char c) 
{
    return isupper(c) ? tolower(c) : toupper(c);
}
