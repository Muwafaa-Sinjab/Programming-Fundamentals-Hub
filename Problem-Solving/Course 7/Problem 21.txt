#include <iostream>
#include <iomanip>
using namespace std;

void PrintFibonacciSeries(int Number);

int main()
{

    PrintFibonacciSeries(10);
    return 0;
}

void PrintFibonacciSeries(int Number)
{
    int Prev1 = 1,  Prev2 = 0;
    int FebNumber = 0;

    cout << setw(2) << "1  ";

    for (int i = 2; i <= Number; i++)
    {
        FebNumber = Prev1 + Prev2;
        cout <<  setw(2) << FebNumber << "  ";
        Prev2 = Prev1;
        Prev1 = FebNumber;
    }
}
