#include <iostream>
#include <iomanip>
using namespace std;

int RandomNumber(int From, int To)
{
    int randNum = rand() % (To - From + 1) + From;
    return randNum;
}

void Print2DArray(int Arr[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        {
            for (int j = 0; j < Cols; j++)
            {
                cout << setw(3) << Arr[i][j] << "  ";
            }
            cout << endl;
        }
}


void Fill2DArray(int Arr[3][3], short Rows, short Cols)
{
    for (int i = 0; i < Rows; i++)
        {
            for (int j = 0; j < Cols; j++)
            {
                Arr[i][j] =  RandomNumber(1, 100);
            }
        }
}

int main() 
{   
    srand((unsigned)time(NULL));

    int Arr[3][3];

    Fill2DArray(Arr, 3, 3);
    cout << "The following is 3x3 random matrix: " << endl;
    Print2DArray(Arr, 3, 3);
    
    return 0;
}
