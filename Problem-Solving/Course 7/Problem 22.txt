#include <iomanip>
using namespace std;

void PrintFibonacciSeries(int Number, int Prev1 , int  Prev2);

int main()
{

    PrintFibonacciSeries(10, 0 , 1);
    return 0;
}

void PrintFibonacciSeries(int Number, int Prev1 , int Prev2)
{
    int FebNumber = 0;

    if(Number > 0)
    {
        FebNumber = Prev1 + Prev2;
        Prev2 = Prev1;
        Prev1 = FebNumber;
        cout << setw(2) << FebNumber << "  ";
        
        PrintFibonacciSeries(Number - 1, Prev1 , Prev2);
    }
}
