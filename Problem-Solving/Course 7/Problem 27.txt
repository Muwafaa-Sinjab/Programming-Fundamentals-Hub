#include <iostream>
#include <cctype>

using namespace std;

char readChar(string message);
char flipCase(char c); 

int main()
{
    char ch = readChar("Please Enter a Character: ");

    ch = flipCase(ch);
    
    cout << "\nCharacter after inverting case: " << ch << endl;

    return 0;
}

char readChar(string message)
{
    char input;
    cout << message;
    cin >> input;
    return input;
}

char flipCase(char c) 
{
    return isupper(c) ? tolower(c) : toupper(c);
}
