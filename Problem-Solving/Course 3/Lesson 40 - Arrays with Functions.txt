#include <iostream>
using namespace std;

void PrintAvg(float Avg){
    
    cout << "Your Average Grade is: " << Avg << endl;
}

float AvgCalculation(float Grade[3]){
    
    return  ( Grade[0] + Grade[1] + Grade[2] ) / 3;
}

void ReadGrades(float Grade[3]){
    
    cout << "This App Will Calculate Your Average of your Grade." << endl;
    cout << "Please Enter Your First Grade: ";
    cin >> Grade[0];
    
    cout << "Please Enter Your second Grade: ";
    cin >> Grade[1];
    
    cout << "Please Enter Your third Grade: ";
    cin >> Grade[2];
    
}


int main()
{
    float Grade[3];
    
    
    ReadGrades(Grade);
    float Avg = AvgCalculation(Grade);
    PrintAvg(Avg);
   
    
    return 0;
}
