#include <iostream>
using namespace std;

int main() {

    float Grade[3];
    float Avg;

    cout << "This App Will Calculate Your Average of your Grades." << endl;

    cout << "Please Enter Your First Grade: ";
    cin >> Grade[0];

    cout << "Please Enter Your Second Grade: ";
    cin >> Grade[1];

    cout << "Please Enter Your Third Grade: ";
    cin >> Grade[2];

    Avg = (Grade[0] + Grade[1] + Grade[2]) / 3;

    cout << "Your Average Grade is: " << Avg << endl;

    return 0;
}
