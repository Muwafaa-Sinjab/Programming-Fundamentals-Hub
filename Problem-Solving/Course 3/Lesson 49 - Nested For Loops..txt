#include <iostream>
using namespace std;


void PrintStarTriangle(){
    
    for (int i = 10; i >= 1; i--){
        
            for(int j = 1; j <= i; j++){
                
                cout << "*";
            }
        cout << "\n";
    }
}

int main()
{
    PrintStarTriangle();
    
    return 0;
}

#include <iostream>
using namespace std;


void PrintNumOrder(){
    
    for (int i = 1; i <= 10; i++){
        
            for(int j = i; j <= 10; j++){
                
                cout << j;
            }

        cout << "\n";
    }
}

int main()
{
    PrintNumOrder();
    
    return 0;
}
