#include <iostream>
using namespace std;

int main()
{
    int Number = 0;
    string Result = "";

    cout << "\n";
    Result = (Number > 0) ? "Positive" : "Negative";
    cout << "Number is " << Result << endl;

    cout << "\n";

    Result = (Number == 0) ? "Zero" : ((Number > 0) ? "Positive" : "Negative");
    cout << "Number is " << Result << endl;
    cout << "\n";
    return 0;
}
