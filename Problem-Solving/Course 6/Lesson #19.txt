#include <iostream>
using namespace std;

void PrintNumbers(int M, int N)
{

    if(M >= N)
    {
        cout << M << endl;
        PrintNumbers(M - 1, N);
    }

}

int main()
{
    PrintNumbers(10, 1);
    return 0;
}

#include <iostream>
using namespace std;

int Recursion(int Base, int Power)
{
    return(Power==0) ? 1 : Base * Recursion(Base,Power - 1);
}

int main() 
{
    cout << Recursion(10, 4);
    return 0;
}
