#include <iostream>
#include <vector>
#include <limits> 
using namespace std;

struct stEmployee
{
    string FirstName = "";
    string LastName = "";
    short Salary = 0;
};

void PrintVector_EmployeeData(vector <stEmployee> &vEmployee)
{
    for(stEmployee &Employee : vEmployee)
    {
        cout << Employee.FirstName << endl;
        cout << Employee.LastName << endl;
        cout << Employee.Salary << endl;
        cout << "-----------------\n";
    }
}

void ReadEmployeeData(vector <stEmployee> &vEmployee)
{
    stEmployee TempEmployee;
    char ReadMore = 'Y';

    while (ReadMore == 'Y' || ReadMore == 'y')
    {
        cout << "First Name: ";
        getline(cin, TempEmployee.FirstName);

        cout << "Last Name: ";
        getline(cin, TempEmployee.LastName);

        cout << "Salary: ";
        cin >> TempEmployee.Salary;
        
        cout << endl;

        vEmployee.push_back(TempEmployee);

        cout << "Do you want to add more employee? Y-N: ";
        cin >> ReadMore;

        cout << endl;
        cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
    }
}

int main() 
{   
    vector <stEmployee> vEmployee;

    ReadEmployeeData(vEmployee);
    PrintVector_EmployeeData(vEmployee);
    
    return 0;
}
