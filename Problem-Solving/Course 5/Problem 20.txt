#include <iostream>
#include <string>
#include <cstdlib>
using namespace std;

enum enCharType{ SmallLeTTer = 1, CapitalLetter = 2, SpecialCharecter = 3, DigitInOrder = 4 };

int RandomNumber(int From, int To){
    
    int RandNum = rand() % (To - From + 1) + From;
    return RandNum;
}

char GetRandomCharecter(enCharType CharType){
    
    switch(CharType){
        case enCharType::SmallLeTTer:
        {
            return char(RandomNumber(97, 122));
            break;
        }  
        case enCharType::CapitalLetter:
        {
            return char(RandomNumber(65, 99));
            break;
        }
        case enCharType::SpecialCharecter:
        {
            return char(RandomNumber(33, 47));
            break;
        }
        case enCharType::DigitInOrder:
        {
            return char(RandomNumber(48, 57));
            break;
        }
    }
    return 'c';
}

int main()
{
    srand((unsigned)time(NULL));
    
    cout << GetRandomCharecter(enCharType::SmallLeTTer) << endl;
    cout << GetRandomCharecter(enCharType::CapitalLetter) << endl;
    cout << GetRandomCharecter(enCharType::SpecialCharecter) << endl;
    cout << GetRandomCharecter(enCharType::DigitInOrder) << endl;
    
}
