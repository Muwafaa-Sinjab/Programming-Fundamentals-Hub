#include <iostream>
#include <string>
using namespace std;

int ReadPositiveNumber(string Message){
    
    int Number = 0;
    do
    {
        cout << Message;
        cin >> Number;
    } while (Number <= 0);

    return Number;
}

int PrintReversedNumber(int Number){
    
    int Reverse = 0;
    int Remainder = 0;
    do{
        
        Remainder = Number % 10;
        Number = Number / 10;
        Reverse = Reverse * 10 + Remainder;
        
    }while(Number > 0);
    
    return Reverse;
}

int main()
{
    
    int Number = ReadPositiveNumber("Please enter a positive number: ");
    int Reverse = PrintReversedNumber(Number);
    cout << "The Reverse of the Number is: " << PrintReversedNumber(Number);
    
    return 0;
}
