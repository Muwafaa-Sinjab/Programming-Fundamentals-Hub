#include <iostream>
#include <string>
using namespace std;

long long ReadPositiveNum(string Massage){
    
    int Number = 0;
    do{
        
        cout << Massage;
        cin >> Number;
        
    }while (Number <= 0);
    
    return Number;
}

int PrintDigits(int Number, short DigitCheck){
    
    int Reminder = 0;
    int Total = 0;
    
    while (Number > 0){
        
        Reminder = Number % 10;
        Number = Number / 10;
        
        if(Reminder == DigitCheck)
            Total++;
       
    }
    
    return Total;
    
}


int main()
{
    long long Read = ReadPositiveNum("Enter Positive Number: ");
    short DigitCheck = ReadPositiveNum("Enter the digit to check: ");
    
    int Total = PrintDigits(Read, DigitCheck);
    
    cout << "\nDigit " << DigitCheck << " Frequency is " << Total << " Time(s)." << endl;
    return 0;
}
