#include <iostream>
#include <string>
using namespace std;

int ReadPositiveNumber(string Message){
    
    int Number = 0;
    do
    {
        cout << Message;
        cin >> Number;
    } while (Number <= 0);

    return Number;
}

int ReversedOrder(int Number){
    
    int Sum = 0;
    int Remainder = 0;
    do{
        
        Remainder = Number % 10;
        Number = Number / 10;
        Sum = Sum + Remainder;
        
    }while(Number > 0);
    
    return Sum;
}

void PrintSumOfReversedOrder(int Sum){
    
    cout << "Sum of Digits is: " << Sum << endl;
}


int main()
{
    
    int Number = ReadPositiveNumber("Please enter a positive number: ");
    int Sum = ReversedOrder(Number);
    PrintSumOfReversedOrder(Sum);
    
    return 0;
}
