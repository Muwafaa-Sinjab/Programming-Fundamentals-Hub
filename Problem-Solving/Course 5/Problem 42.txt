#include <iostream>
using namespace std;

int RandomNumber(int From, int To)
{
    int randNum = rand() % (To - From + 1) + From;
    return randNum;
}

void FillArratWithRandomNumbers(int Arr[100], int& ArrLength)
{
    cout << "Enter Number of Elements: ";
    cin >> ArrLength;

    for (int i = 0; i < ArrLength; i++)
    {
        Arr[i] = RandomNumber(1, 100);
    }
}

void PrintArrayElemnts(int Arr[100], int ArrLength)
{
    for (int i = 0; i < ArrLength; i++)
    {
        cout << Arr[i] << " ";
    }
    cout << endl;
}

int CountOddNumbers(int Arr[100], int ArrLength)
{
    int NumberOfOdd = 0;

    for (int i = 0; i < ArrLength; i++)
    {
        if(Arr[i] % 2 != 0)
        {
            NumberOfOdd++;
        }
    }
    return NumberOfOdd;
}

int main()
{
    srand((unsigned)time(NULL));

    int Arr[100], ArrLength = 0;
    
    FillArratWithRandomNumbers(Arr, ArrLength);

    cout << "Array Elements:\n";
    PrintArrayElemnts(Arr, ArrLength);

    cout << "\nOdd Number cout is: " << CountOddNumbers(Arr, ArrLength);

    return 0;
}
