#include <iostream>
using namespace std;

int main()
{
    int A, B;
    
    cout<<"Please Enter the first Number (A):" << endl;
    cin >> A;
    cout<<"Please Enter the first Number (B):" << endl;
    cin >> B;
    
    cout << "\n";
    
    cout << "A = B is " << (A == B) << endl;
    cout << "A != B is " << (A != B) << endl;
    cout << "A < B is " << (A < B) << endl;
    cout << "A > B is " << (A > B) << endl;
    cout << "A <= B is " << (A <= B) << endl;
    cout << "A >= B is " << (A >= B) << endl;
    

    return 0;
}
